// App.js
import React, { useEffect, useRef, useState } from 'react';
import { StyleSheet, View } from 'react-native';
import { Accelerometer } from 'expo-sensors';
import { WebView } from 'react-native-webview';

export default function App() {
  const webviewRef = useRef(null);
  const [data, setData] = useState({ x:0, y:0, z:0 });

  useEffect(() => {
    // sample 4×/s
    Accelerometer.setUpdateInterval(250);
    const sub = Accelerometer.addListener(acc => {
      setData(acc);
      // push to WebView
      webviewRef.current?.postMessage(JSON.stringify(acc));
    });
    return () => sub.remove();
  }, []);

  // the HTML + JS runtime for Magenta.js + Tone.js
  const html = `
<!DOCTYPE html>
<html>
<head>
  <meta charset="utf-8"/>
  <style>body,html{margin:0;padding:0;background:#000}</style>
  <script src="https://cdn.jsdelivr.net/npm/@magenta/music@1.28.0"></script>
  <script src="https://cdn.jsdelivr.net/npm/tone@14.7.77/build/Tone.min.js"></script>
</head>
<body></body>
<script>
(async function(){
  await Tone.start();
  // poly synth for drums
  const synth = new Tone.PolySynth({
    voice: Tone.MembraneSynth
  }).toDestination();

  // load the tiny drum_rnn model
  const rnn = new mm.MusicRNN(
    'https://storage.googleapis.com/magentadata/js/checkpoints/music_rnn/drum_kit_rnn'
  );
  await rnn.initialize();

  // we’ll keep one seed and just continue it each time
  const seed = {
    quantizationInfo: {stepsPerQuarter:1},
    totalQuantizedSteps:4,
    notes: [
      {pitch:36, quantizedStartStep:0, quantizedEndStep:1},
      {pitch:38, quantizedStartStep:2, quantizedEndStep:3}
    ]
  };

  // listen for sensor messages
  window.document.addEventListener('message', async (e) => {
    const d = JSON.parse(e.data);
    // map acceleration magnitude to tempo [80–180] and temperature [0.1–1.5]
    const a = Math.sqrt(d.x*d.x + d.y*d.y + d.z*d.z);
    const tempo = Math.min(Math.max(80 + a*80, 80), 180);
    const temp  = Math.min(Math.max((a - 0.5)*2, 0.1), 1.5);

    Tone.Transport.bpm.value = tempo;

    // generate a 4-bar continuation (16 steps)
    const continuation = await rnn.continueSequence(seed, 16, temp);
    // schedule & play immediately
    Tone.Transport.cancel();    
    continuation.notes.forEach(n => {
      const t0 = (n.quantizedStartStep / 4) + 'm';
      const dur = ((n.quantizedEndStep - n.quantizedStartStep) / 4) + 'm';
      const freq = Tone.Frequency(n.pitch, 'midi').toFrequency();
      synth.triggerAttackRelease(freq, dur, Tone.now() + Tone.Time(t0).toSeconds());
    });
    // restart transport so scheduling happens
    Tone.Transport.start();
  });
})();
</script>
</html>
`;

  return (
    <View style={styles.container}>
      <WebView
        ref={webviewRef}
        originWhitelist={['*']}
        source={{ html }}
        javaScriptEnabled
        domStorageEnabled
        style={styles.webview}
      />
    </View>
  );
}

const styles = StyleSheet.create({
  container: { flex:1, backgroundColor:'#000' },
  webview: { flex:1 }
});
